## Настройка подключения к MySQL
## Используется для статистики, сбора отзывов,
## администрирования и работы с библиотекой игр.

## Должен ли быть включен SQL? Раскомментируйте, чтобы включить
SQL_ENABLED

## Сервер, на котором находится база данных MySQL, можно найти по адресу.
## Примеры: localhost, 200.135.5.43, www.mysqldb.com и т.д.
ADDRESS 127.0.0.1

## Порт сервера MySQL (по умолчанию - 3306).
PORT 3306

## База данных для всех функций SQL, а не только для обратной связи.
FEEDBACK_DATABASE andromeda_schema

## Префикс, который должен быть добавлен к имени каждой таблицы, в старых базах данных для этого потребуется установить значение erro_
## Обратите внимание, что это не изменяет имена таблиц в базе данных, вам придется сделать это самостоятельно.
##IE:
##	FEEDBACK_TABLEPREFIX
##	FEEDBACK_TABLEPREFIX SS13_
## Удалите "SS13_", если вы используете стандартный файл схемы.
FEEDBACK_TABLEPREFIX

## Имя пользователя/логин, используемый для доступа к базе данных.
FEEDBACK_LOGIN root

## Пароль, используемый для доступа к базе данных.
FEEDBACK_PASSWORD root

## Время ожидания в секундах для асинхронных запросов.
## Устанавливается равным 0 для бесконечности.
ASYNC_QUERY_TIMEOUT 10

## Время в секундах, необходимое для выполнения блокирующих запросов до истечения времени ожидания медленного запроса
## Установите значение 0 для бесконечности
## Должно быть меньше или равно ASYNC_QUERY_TIMEOUT
BLOCKING_QUERY_TIMEOUT 5

## Минимальное количество sql-подключений, которое необходимо поддерживать в пуле. Увеличение этого значения на серверах, географически удаленных от базы данных, может повысить производительность.
POOLING_MIN_SQL_CONNECTIONS 1

## Максимальное количество sql-подключений к базе данных.
POOLING_MAX_SQL_CONNECTIONS 25

## Максимальное количество одновременных асинхронных запросов, которые могут ожидать результата, прежде чем мы начнем ставить в очередь дальнейшие запросы к базе данных.
MAX_CONCURRENT_QUERIES 25
